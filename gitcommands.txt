ssh-keygen -t rsa -C "Alesyrex@gmail.com"  -  сгенерировать ssh key пару, -С - комментарий, потом public key вставляется в github
ls - посмотреть все директории и файлы в текущей директории
touch "file" - создать файл (без ковычек если не полный путь с пробелами)
start "file" - открыть файл
rm "file" - удалить файл
echo "text" >> "file" - дописать в конец файла текст (без ковычек)
git config --global user.name "Name"  -  задать имя пользователя
git config --global user.email "Email"  - задать почту
git config --global core.editor "'C:/Program Files/Notepad++/notepad++.exe'"  -  задать текстовый редактор по умолчанию
git config --edit  -  редактировать .git/config
git clone - скачать новый удаленный репозиторий
git init - создать репозиторий в данной папке (создает папку .git и ветку мастер)
git status - посмотреть какие есть текущие изменения
git add somefile.txt - добавить файл с изменениями в индекс (перед коммитом)
git add . - добавить все файлы с изменениями в индекс
git commit -m "comment"  -  сделать коммит с уомментарием в кавычках
git log - посмотреть изменения внесенные в репозиторий (сколько и какие коммиты), с флагом -p покажет с подробностями
git show 1af17e  - покажет полный список изменений в конкретном коммите (указываются первые несколько символов sha1 коммита)
git diff - показывает изменения внесенные в репозиторий, не подготовленные для фиксации, с флагом --staged подготовленные
git checkout somefile.txt - восстановил указанный файл, не находящийся в индексе к предыдущему значению
git restore somefile.txt - удалит указанный файл ненаходящийся в индексе
git restore --staged - уберет из индекса указанный файл в неподготовленное состояние
git commit --amend -m "I forgot some changes"  -  внести изменения в файлах в предыдущий коммит
git revert HEAD  -  откатывает последний коммит к предыдущему состоянию (с указанием вместо HEAD определенного коммита, возвращает к состоянию того коммита)
git branch newbranchname - создает новую ветку с указанным именем
git branch  -  покажет все ветки
git branch -d namebranch - удалит указанную ветки (сделает только после слияния с другой веткой)
git push origin --delete namebranch - удалить ветку на удаленном репозитории
git merge namebranch - объеденит указанную ветку с основной
git merge --abort - отменить процесс слияния
git remote -v  -  посмотреть какие удаленные репозитории
get remote remove origin  -  отвязать удаленный репозиторий от данного локального
git remote add origin http://github.com/someurl..  -  добавить удаленный репозиторий (привязать к текущему локальному)
git push  -  отправить изменения в удаленный репозиторий
git push origin master  -  отправить изменения в конкретную ветку удаленного репозитория
git push --set-upstream origin master  -  загрузить изменения на удаленный репозиторий с переключением ветки на удаленном репозитории (наприм на ветку master)
git push -u origin main - загрузить на пустой репозиторий гитхаба (если там нет даже веток)
git pull - загрузка изменений с удаленного репозитория (с флагом --verbose покажет подробности)
git rebase namebranch - передвигает головной коммит master ветки если там появились еще коммиты
git branch --move --force main master  -  переименовать ветку, наприм с main на master
git tag tagname  -  тегировать последний коммит
giе fetch - скачитвает все изменения с удаленного репозитория не сливая их с локальным, далее выполняем git branch -r, git diff - смотрим изменения по сравнению с локальным репозиторием, git checkout - переключаемся на удаленную ветку, git merge - делаем слияние с удаленной веткой
git gui& - git с графическим интерфейсом (простейшие комманды делать удобнее)
gitk&  -  графическое отображение веток с доп информацией
git remote set-url origin https://token@github.com/Alesyrex/Java-tasks - для авторизации на гитхабе (вместо token указывай свой токен)
